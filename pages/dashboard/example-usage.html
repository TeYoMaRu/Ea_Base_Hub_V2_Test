วิธีการใช้งาน

  1. โหลด Scripts ตามลำดับใน HTML: 
    <!-- 1. Supabase CDN -->
<script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2"></script>

<!-- 2. Supabase Client Config -->
<script src="supabaseClient.js"></script>

<!-- 3. Auth Service -->
<script src="auth.js"></script>

<!-- 4. User Service -->
<script src="userService.js"></script>
  2. ป้องกันหน้า (Page Protection): 
    // ป้องกันหน้า - ต้อง login
document.addEventListener('DOMContentLoaded', async () => {
  await protectPage(); // ทุกคนที่ login ได้เข้าได้
});

// ป้องกันหน้า - เฉพาะ admin
document.addEventListener('DOMContentLoaded', async () => {
  await protectPage(['admin']); // เฉพาะ admin
});

// ป้องกันหน้า - admin หรือ manager
document.addEventListener('DOMContentLoaded', async () => {
  await protectPage(['admin', 'manager']);
});
  3. Auto-fill ฟอร์ม: 
    // หลัง initUserService() เสร็จแล้ว
autoFillUserData({
  full_name: "formFullName",
  email: "formEmail",
  zone: "formZone",
  phone: "formPhone",
  readonly: ["full_name", "email"]
});
  4. ดึงข้อมูล User: 
    const userName = getUserData("full_name");
const userZone = getUserData("zone");
const allUserData = getUserData();
